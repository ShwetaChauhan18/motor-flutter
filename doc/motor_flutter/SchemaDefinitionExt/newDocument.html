SchemaDocument newDocument(String label) {
  var docFields = <SchemaDocumentValue>[];
  for (SchemaField f in fields) {
    final key = f.name;
    final kind = f.fieldKind;
    docFields.add(SchemaDocumentValue(key: key, kind: kind.kind));
  }
  return SchemaDocument(
    label: label,
    schemaDid: did,
    fields: docFields,
  );
}
